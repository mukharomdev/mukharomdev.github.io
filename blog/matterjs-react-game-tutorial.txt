f:T1aa2,<p>Untuk membuat game menggunakan <strong>Matter.js</strong> dan <strong>React</strong>, Anda dapat membagi aplikasi menjadi beberapa komponen React untuk memisahkan tanggung jawab dan menjaga kode tetap terorganisir. Berikut adalah contoh struktur komponen yang dapat digunakan:</p>
<hr>
<h3><strong>Struktur Komponen untuk Game Matter.js</strong></h3>
<ol>
<li>
<p><strong>App</strong> (Komponen Utama):</p>
<ul>
<li>Bertanggung jawab untuk menginisialisasi game dan merender komponen lainnya.</li>
<li>Mengelola state global jika diperlukan (misalnya, skor, status game).</li>
</ul>
</li>
<li>
<p><strong>GameWorld</strong> (Komponen Dunia Game):</p>
<ul>
<li>Bertanggung jawab untuk menginisialisasi dan mengelola dunia fisika Matter.js.</li>
<li>Menangani logika game seperti menambahkan objek, mendeteksi tabrakan, dan merender canvas.</li>
</ul>
</li>
<li>
<p><strong>Player</strong> (Komponen Pemain):</p>
<ul>
<li>Bertanggung jawab untuk mengelola objek pemain (misalnya, karakter atau kendaraan).</li>
<li>Menangani input pengguna (misalnya, keyboard atau mouse) untuk mengontrol pemain.</li>
</ul>
</li>
<li>
<p><strong>Obstacle</strong> (Komponen Rintangan):</p>
<ul>
<li>Bertanggung jawab untuk membuat dan mengelola rintangan atau objek statis di dunia game.</li>
</ul>
</li>
<li>
<p><strong>Scoreboard</strong> (Komponen Skor):</p>
<ul>
<li>Bertanggung jawab untuk menampilkan skor atau status game.</li>
</ul>
</li>
<li>
<p><strong>GameControls</strong> (Komponen Kontrol):</p>
<ul>
<li>Bertanggung jawab untuk menampilkan tombol atau kontrol game (misalnya, mulai, jeda, ulang).</li>
</ul>
</li>
</ol>
<hr>

<h4>1. <strong>App (Komponen Utama)</strong></h4>
<pre><code class="language-javascript">import React, { useState } from 'react';
import GameWorld from './GameWorld';
import Scoreboard from './Scoreboard';
import GameControls from './GameControls';

function App() {
  const [score, setScore] = useState(0);
  const [isGameRunning, setIsGameRunning] = useState(false);

  const startGame = () => {
    setIsGameRunning(true);
    setScore(0);
  };

  const endGame = () => {
    setIsGameRunning(false);
  };

  return (
    &#x3C;div>
      &#x3C;h1>Matter.js Game&#x3C;/h1>
      &#x3C;Scoreboard score={score} />
      &#x3C;GameControls
        isGameRunning={isGameRunning}
        onStart={startGame}
        onEnd={endGame}
      />
      {isGameRunning &#x26;&#x26; &#x3C;GameWorld setScore={setScore} />}
    &#x3C;/div>
  );
}

export default App;

<h4>2. <strong>GameWorld (Komponen Dunia Game)</strong></h4>
<pre><code class="language-javascript">import React, { useEffect, useRef } from 'react';
import Matter from 'matter-js';
import Player from './Player';
import Obstacle from './Obstacle';

function GameWorld({ setScore }) {
  const sceneRef = useRef(null);

  useEffect(() => {
    const { Engine, Render, Runner, World, Bodies } = Matter;

    // Buat engine dan renderer
    const engine = Engine.create();
    const render = Render.create({
      element: sceneRef.current,
      engine: engine,
      options: {
        width: 800,
        height: 600,
        wireframes: false,
        background: '#f4f4f4',
      },
    });

    // Tambahkan canvas ke DOM
    sceneRef.current.appendChild(render.canvas);

    // Buat dunia fisika
    const ground = Bodies.rectangle(400, 590, 810, 60, { isStatic: true });
    World.add(engine.world, [ground]);

    // Tambahkan pemain dan rintangan
    const player = Player(engine);
    const obstacle = Obstacle(engine);

    // Deteksi tabrakan untuk skor
    Matter.Events.on(engine, 'collisionStart', (event) => {
      event.pairs.forEach((pair) => {
        if (pair.bodyA === player || pair.bodyB === player) {
          setScore((prevScore) => prevScore + 1);
        }
      });
    });

    // Jalankan renderer dan engine
    Render.run(render);
    const runner = Runner.create();
    Runner.run(runner, engine);

    // Bersihkan saat komponen di-unmount
    return () => {
      Render.stop(render);
      Runner.stop(runner);
      World.clear(engine.world, false);
      Engine.clear(engine);
      render.canvas.remove();
    };
  }, [setScore]);

  return &#x3C;div ref={sceneRef}>&#x3C;/div>;
}

export default GameWorld;
</code></pre>

<h4>3. <strong>Player (Komponen Pemain)</strong></h4>
<pre><code class="language-javascript">import Matter from 'matter-js';

function Player(engine) {
  const player = Matter.Bodies.rectangle(100, 100, 50, 50, {
    restitution: 0.9,
    render: {
      fillStyle: 'blue',
    },
  });

  Matter.World.add(engine.world, [player]);

  // Tambahkan kontrol keyboard
  document.addEventListener('keydown', (event) => {
    if (event.key === 'ArrowRight') {
      Matter.Body.setVelocity(player, { x: 5, y: 0 });
    }
    if (event.key === 'ArrowLeft') {
      Matter.Body.setVelocity(player, { x: -5, y: 0 });
    }
  });

  return player;
}

export default Player;
</code></pre>

<h4>4. <strong>Obstacle (Komponen Rintangan)</strong></h4>
<pre><code class="language-javascript">import Matter from 'matter-js';

function Obstacle(engine) {
  const obstacle = Matter.Bodies.rectangle(400, 300, 80, 80, {
    isStatic: true,
    render: {
      fillStyle: 'red',
    },
  });

  Matter.World.add(engine.world, [obstacle]);

  return obstacle;
}

export default Obstacle;
